 package scenario1;
 
 
 import java.util.ArrayList;
 import java.util.List;
 
 
 class CountUpTo {
 
     public static List<Object> countUpTo(int n) {
         List<Object> result = new ArrayList<Object>();
         int i = 2;
         while (i < n) {
             if (isPrime(i)) {
                 result.add(i);
             }
             i++;
         }
         return result;
     }
 
     private static boolean isPrime(int n) {
         int i = 2;
         while (i < n) {
             if (n % i == 0) {
                 return false;
             }
             i++;
         }
         return true;
     }
 
 
 }
