 package scenario2;
 
 
 class XOrY {
     /**
      * A simple program which should return the value of x if n is 
      * a prime number and should return the value of y otherwise.
      */
     public static int xOrY(int n, int x, int y) {
         if (n == 1) {
             return y;
         }
         for (int i = 2; i < n; i++) {
             if (n % i == 0) {
                 return y;
             }
         }
         return x;
     }
 
 }
