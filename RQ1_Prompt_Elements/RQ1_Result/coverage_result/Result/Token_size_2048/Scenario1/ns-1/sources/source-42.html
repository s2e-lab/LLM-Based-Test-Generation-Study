 package scenario1;
 
 
 class HowManyTimes {
 
     public static int howManyTimes(String string, String substring) {
         int count = 0;
         int index = 0;
         while (index < string.length()) {
             index = string.indexOf(substring, index);
             if (index == -1) {
                 break;
             }
             count++;
             index++;
         }
         return count;
     }
